        // Update connection status indicator

        function updateConnectionStatus(status, message) {

            const statusEl = document.getElementById('connectionStatus');

            statusEl.className = `connection-status ${status}`;

            

            switch(status) {

                case 'connected':

                    statusEl.textContent = '✓ Live Data Connected';

                    break;

                case 'disconnected':

                    statusEl.textContent = '✗ Connection Failed';

                    break;

                case 'loading':

                    statusEl.textContent = '🔄 Loading Dashboard Data...';

                    break;

            }

        }


        // Load all dashboard data from live APIs

        async function loadDashboardData() {

            console.log('[Dashboard] Loading REAL data from Supabase API...');

            updateConnectionStatus('loading');

            

            let successCount = 0;

            let totalCalls = 4;


            try {

                // Load Sprint Tasks

                try {

                    const tasksResponse = await fetch(`${API_BASE}/api/tasks`);

                    if (tasksResponse.ok) {

                        const tasksData = await tasksResponse.json();

                        dashboardData.tasks = tasksData.data?.tasks || tasksData.tasks || [];

                        updateTasksDisplay();

                        successCount++;

                    }

                } catch (error) {

                    console.error('[Dashboard] Tasks API error:', error);

                }


                // Load Podcast Calls

                try {

                    const podcastResponse = await fetch(`${API_BASE}/api/podcast-interviews`);

                    if (podcastResponse.ok) {

                        const podcastData = await podcastResponse.json();

                        dashboardData.podcastCalls = podcastData.data?.interviews || podcastData.interviews || [];

                        updatePodcastDisplay();

                        successCount++;

                    }

                } catch (error) {

                    console.error('[Dashboard] Podcast API error:', error);

                }


                // Load Discovery Calls

                try {

                    const discoveryResponse = await fetch(`${API_BASE}/api/discovery-calls`);

                    if (discoveryResponse.ok) {

                        const discoveryData = await discoveryResponse.json();

                        dashboardData.discoveryCalls = discoveryData.data?.calls || discoveryData.calls || [];

                        updateDiscoveryDisplay();

                        successCount++;

                    }

                } catch (error) {

                    console.error('[Dashboard] Discovery API error:', error);

                }


                // Load Pipeline

                try {

                    const pipelineResponse = await fetch(`${API_BASE}/api/pipeline`);

                    if (pipelineResponse.ok) {

                        const pipelineData = await pipelineResponse.json();

                        dashboardData.pipeline = pipelineData.data?.stages || pipelineData.stages || [];

                        updatePipelineDisplay();

                        successCount++;

                    }

                } catch (error) {

                    console.error('[Dashboard] Pipeline API error:', error);

                }


                // Update connection status

                if (successCount > 0) {

                    updateConnectionStatus('connected');

                    document.getElementById('api-status').innerHTML = '<span style="color: #27ae60;">All Systems Connected</span>';

                } else {

                    updateConnectionStatus('disconnected');

                    document.getElementById('api-status').innerHTML = '<span style="color: #e74c3c;">Connection Failed</span>';

                }

                const now = new Date();

                document.getElementById('last-update-time').textContent = now.toLocaleString();

                    discoveryCalls: dashboardData.discoveryCalls.length,

                    pipeline: dashboardData.pipeline.length,

                    successRate: `${successCount}/${totalCalls}`

                });

                updateConnectionStatus('disconnected');

                document.getElementById('api-status').innerHTML = '<span style="color: #e74c3c;">Connection Error</span>';

            }

        }


        // Update Tasks Display

        function updateTasksDisplay() {

            const tasks = dashboardData.tasks;

            

            // Count by status

            const finished = tasks.filter(t => t.status === 'completed' || t.status === 'finished').length;

            const onTrack = tasks.filter(t => t.status === 'on-track' || t.status === 'in-progress').length;

            const offTrack = tasks.filter(t => t.status === 'off-track').length;

            const blocked = tasks.filter(t => t.status === 'blocked').length;

            

            // Update stat cards

            document.getElementById('tasks-finished').textContent = finished;

            document.getElementById('tasks-on-track').textContent = onTrack;

            document.getElementById('tasks-off-track').textContent = offTrack;

            document.getElementById('tasks-blocked').textContent = blocked;

            

            // Update tasks completed ratio

            const total = tasks.length;

            const percentage = total > 0 ? Math.round((finished / total) * 100) : 0;

            document.getElementById('tasks-completed-ratio').textContent = `${finished}/${total} (${percentage}%)`;

            

            // Render task list

            const tasksContainer = document.getElementById('sprint-tasks');

            if (tasks.length === 0) {

                tasksContainer.innerHTML = '<div style="text-align: center; padding: 2rem; color: #7f8c8d;">No tasks found</div>';

                return;

            }

            

            tasksContainer.innerHTML = tasks.slice(0, 5).map(task => `

                <div class="task-item">

                    <div class="task-details">

                        <h4>${task.title || 'Untitled Task'}</h4>

                        <p>${task.description || 'No description'}</p>

                    </div>

                    <div class="status-badge status-${task.status}">${task.status}</div>

                </div>

            `).join('');

        }


        // Update Podcast Display

        function updatePodcastDisplay() {

            const calls = dashboardData.podcastCalls;

            

            // Update podcast calls count

            document.getElementById('podcast-calls').textContent = calls.length;

            // Update metrics

            document.getElementById('weekly-calls').textContent = `${calls.length} calls`;

            

            const qualificationRate = calls.length > 0 ? Math.round((qualified / calls.length) * 100) : 0;

            document.getElementById('metric-qualification-rate').textContent = `${qualificationRate}%`;

            

            const avgScore = calls.length > 0 

                ? (calls.reduce((sum, c) => sum + (c.qualificationScore || c.score || 0), 0) / calls.length).toFixed(1)

                : '0.0';

            document.getElementById('metric-avg-score').textContent = avgScore;

            

            document.getElementById('metric-ai-analysis').textContent = '✓';

            document.getElementById('metric-ready-discovery').textContent = qualified;

        }
